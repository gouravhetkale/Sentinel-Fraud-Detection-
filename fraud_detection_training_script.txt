import pandas as pd
import numpy as np
from sklearn.ensemble import RandomForestClassifier
from sklearn.preprocessing import StandardScaler
from imblearn.over_sampling import SMOTE
from sklearn.utils import resample
import joblib
import time

def simulate_loading(message):
    print(message, end="", flush=True)
    for _ in range(3):
        time.sleep(0.4)
        print(".", end="", flush=True)
    print("\n")

def run_training():
    print("--- Sentinel AI: Optimised Aggressive Training Pipeline ---")
    DATA_FILE = "creditcard.csv"
    MODEL_FILE = "sentinel_model.pkl"
    SCALER_FILE = "scaler.pkl"

    try:
        # 1. Load Dataset
        print(f"Reading {DATA_FILE}...")
        df = pd.read_csv(DATA_FILE)
        
        # 2. Global Scaling
        # We fit the scaler on the ENTIRE dataset first to capture the true 
        # distribution of transaction amounts before we resample.
        print("Calibrating global currency scales...")
        scaler = StandardScaler()
        scaler.fit(df['Amount'].values.reshape(-1, 1))
        
        # 3. Smart Resampling (Fix for training hang)
        # To maintain a 50/50 balance without crashing the system:
        fraud = df[df['Class'] == 1]
        genuine = df[df['Class'] == 0]
        
        # We take a larger subset of genuine transactions to provide more variety
        # while still keeping it small enough to train quickly.
        print("Initialising Smart Resampling...")
        genuine_downsampled = resample(genuine, 
                                     replace=False, 
                                     n_samples=60000, 
                                     random_state=42)
        
        # Combine back
        df_balanced = pd.concat([genuine_downsampled, fraud])
        
        # Apply scaling to the working set
        df_balanced['Amount'] = scaler.transform(df_balanced['Amount'].values.reshape(-1, 1))
        
        X = df_balanced.drop(['Class', 'Time'], axis=1)
        y = df_balanced['Class']

        # 4. Apply SMOTE for perfect 50/50 split
        simulate_loading("Applying SMOTE for 50% Fraud / 50% Genuine ratio")
        sm = SMOTE(sampling_strategy=1.0, random_state=42)
        X_res, y_res = sm.fit_resample(X, y)
        
        print(f"✅ Balanced Dataset Created: {len(y_res[y_res==1]):,} Fraud / {len(y_res[y_res==0]):,} Genuine")

        # 5. Train the Random Forest (Hyper-Sensitive Mode)
        simulate_loading("Training Hyper-Sensitive Ensemble (150 Trees)")
        model = RandomForestClassifier(
            n_estimators=150,      # Increased trees for better resolution
            max_depth=None,        # No limit: capture every fraud signature
            min_samples_leaf=1,    # High specificity
            class_weight='balanced', # Extra weight to the minority patterns
            random_state=42, 
            n_jobs=-1
        )
        model.fit(X_res, y_res)
        
        # 6. Save the technical state
        print("💾 Saving technical state (Model & Scaler)...")
        joblib.dump(model, MODEL_FILE)
        joblib.dump(scaler, SCALER_FILE)
        
        print("\n✅ Training Successfully Completed.")
        print("🚀 The system is now tuned for maximum fraud detection.")

    except FileNotFoundError:
        print(f"❌ Error: {DATA_FILE} not found in the root folder.")
    except Exception as e:
        print(f"❌ An error occurred: {e}")

if __name__ == "__main__":
    run_training()
